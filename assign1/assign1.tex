x(n)&= x(0)\,r^{n}
\\x(5)&=x(0)\,r^5
\\x(8)&=x(0)\,r^8
\\x({11})&=x(0)\,r^{11}
\\x(8)\,x(8)&=x(0)\,r^8\,x(0)\,r^8

       \\&=x(0)^2\,r^{16}

\\q^2&=ps
\end{align}
